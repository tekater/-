    Сокращения в регулярных выражениях

        \d - любое число [0-9]
        \s - пробелы [\t \n \x0B \f \r];
        \w - буквенно-числовые символы [a-zA-Z_0-9]
        \b - граница. \ba
        \D не цифра == [^0-9]
        \S не пробел == [^ \t \n \x0B \f \r];
        \W не буквенно-числовой символ [^a-zA-z_0-9]
        \B не граница слова.
